public class pattern
{
public static void main(String arg[])
{
    int n=5;
for(int i=1;i<=n;i++)
{
for(int k=1;k<=i-1;k++)
        {
           System.out.print(" ");
        }
for(int j=1;j<=n;j++)
        {
           System.out.print("* ");
        }
 System.out.println();
}
}
}
